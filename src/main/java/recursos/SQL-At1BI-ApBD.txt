DROP TABLE IF EXISTS Emprestimo;
DROP TABLE IF EXISTS Aluno;
DROP TABLE IF EXISTS Publicacao;

CREATE TABLE Aluno (
    matriculaAluno integer NOT NULL,
    nome varchar(255),
    PRIMARY KEY (matriculaAluno)
);

CREATE TABLE Publicacao (
    codigoPub integer GENERATED BY DEFAULT AS IDENTITY,
    ano integer,
    autor varchar(255),
    tipo varchar(255),
    titulo varchar(255),
    PRIMARY KEY (codigoPub)
);

CREATE TABLE Emprestimo (
    idEmprestimo integer GENERATED BY DEFAULT AS IDENTITY,
    dataDevolucao date,
    dataEmprestimo date,
    id_aluno integer,
    id_publicacao integer,
    PRIMARY KEY (idEmprestimo),
    CONSTRAINT FK_aluno FOREIGN KEY (id_aluno) REFERENCES Aluno(matriculaAluno),
    CONSTRAINT FK_publicacao FOREIGN KEY (id_publicacao) REFERENCES Publicacao(codigoPub)
);
